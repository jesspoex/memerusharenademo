"use client";

import { createContext, useContext, useState, useEffect, useCallback, ReactNode, useRef } from 'react';
import { 
  Round, Player, PlayerBet, RoundResult, GameState, GameConfig, Token,
  DEFAULT_GAME_CONFIG, AVAILABLE_TOKENS
} from '@/lib/game/types';
import {
  calculatePlatformFee,
  calculateActualBet,
  calculateRewards,
  generateFakePlayers,
  hasRoundEnded,
  calculateTimeLeft,
  sortPlayersByProfit,
  generateRandomTokenPair,
} from '@/lib/game/utils';
import { updateTokensWithRealData } from '@/lib/game/api';

interface GameContextType {
  gameState: GameState;
  currentTime: number;
  timeLeft: number;
  currentPlayer: Player | null;
  placeBet: (amount: number, tokenSide: 'A' | 'B') => boolean;
  getPlayerBets: (wallet: string) => PlayerBet[];
  getLeaderboard: () => Player[];
  getCurrentRoundProgress: () => number;
}

const GameContext = createContext<GameContextType | undefined>(undefined);

interface GameProviderProps {
  children: ReactNode;
  config?: Partial<GameConfig>;
  userWallet?: string | null;
}

export function GameProvider({ children, config: customConfig, userWallet }: GameProviderProps) {
  const config = { ...DEFAULT_GAME_CONFIG, ...customConfig };
  const [currentTime, setCurrentTime] = useState(Date.now());
  const [gameState, setGameState] = useState<GameState>(() => initializeGameState(config));
  const gameLoopRef = useRef<NodeJS.Timeout | null>(null);
  const priceUpdateRef = useRef<NodeJS.Timeout | null>(null);

  function initializeGameState(cfg: GameConfig): GameState {
    const [tokenA, tokenB] = generateRandomTokenPair(AVAILABLE_TOKENS);
    const now = Date.now();
    
    return {
      currentRound: {
        id: 1,
        status: 'waiting',
        startTime: now,
        endTime: null,
        tokens: [tokenA, tokenB],
        tokenAPool: 0,
        tokenBPool: 0,
        prizePool: 0,
      },
      players: [],
      roundHistory: [],
      config: cfg,
    };
  }

  // Update current time every second
  useEffect(() => {
    const interval = setInterval(() => {
      setCurrentTime(Date.now());
    }, 1000);
    return () => clearInterval(interval);
  }, []);

  // Fetch real prices every 5 seconds
  useEffect(() => {
    const updatePrices = async () => {
      try {
        const tokens = gameState.currentRound.tokens;
        const updated = await updateTokensWithRealData(tokens);
        const [tokenA, tokenB] = updated;
        
        setGameState(prev => ({
          ...prev,
          currentRound: {
            ...prev.currentRound,
            tokens: [tokenA, tokenB],
          },
        }));
      } catch (error) {
        console.error('Price update error:', error);
      }
    };

    updatePrices();
    priceUpdateRef.current = setInterval(updatePrices, 5000);
    
    return () => {
      if (priceUpdateRef.current) clearInterval(priceUpdateRef.current);
    };
  }, [gameState.currentRound.tokens]);

  // ✅ FORCE START ROUND - Auto start after 10 seconds or 2+ players
  useEffect(() => {
    if (gameState.currentRound.status === 'waiting') {
      const waitingTime = currentTime - (gameState.currentRound.startTime || currentTime);
      
      // IMMEDIATE START: Start after 10 seconds OR if 2+ players
      if (waitingTime > 10000 || gameState.players.length >= 2) {
        startRound();
      }
    }
  }, [currentTime, gameState.currentRound.status, gameState.players.length]);

  // Main game loop
  useEffect(() => {
    const runGameLoop = () => {
      const { currentRound, players } = gameState;
      const now = Date.now();

      // Start round if tokens have prices and in waiting status
      if (currentRound.status === 'waiting' && currentRound.tokens[0].price > 0 && currentRound.tokens[1].price > 0) {
        const waitingTime = now - (currentRound.startTime || now);
        if (waitingTime > 10000 || players.length >= 2) {
          startRound();
        }
      }
      
      // End round when time is up
      if (currentRound.status === 'live' && currentRound.endTime && hasRoundEnded(currentRound.endTime, now)) {
        endRound();
      }

      // Simulate auto-trading during live round
      if (currentRound.status === 'live') {
        simulateAutoTrading();
      }

      // Add fake players for demo mode
      if (config.fakePlayersEnabled && players.filter(p => !p.isFake).length === 0) {
        addFakePlayers();
      }
    };

    gameLoopRef.current = setInterval(runGameLoop, 1000);
    return () => {
      if (gameLoopRef.current) clearInterval(gameLoopRef.current);
    };
  }, [gameState, currentTime, config.fakePlayersEnabled]);

  // ✅ START ROUND
  const startRound = useCallback(() => {
    const now = Date.now();
    
    setGameState(prev => ({
      ...prev,
      currentRound: {
        ...prev.currentRound,
        status: 'live',
        startTime: now,
        endTime: now + (prev.config.roundDuration * 1000),
        tokenAPool: 0,
        tokenBPool: 0,
        prizePool: 0,
      },
      players: prev.players.map(p => ({ ...p, profit: 0, profitPercentage: 0 })),
    }));
  }, []);

  // ✅ START NEXT ROUND (MUST BE BEFORE endRound)
  const startNextRound = useCallback(() => {
    const [newTokenA, newTokenB] = generateRandomTokenPair(
      AVAILABLE_TOKENS, 
      gameState.currentRound.tokens
    );
    
    const now = Date.now();
    
    setGameState(prev => ({
      ...prev,
      currentRound: {
        id: prev.currentRound.id + 1,
        status: 'waiting',
        startTime: now,
        endTime: null,
        tokens: [newTokenA, newTokenB],
        tokenAPool: 0,
        tokenBPool: 0,
        prizePool: 0,
      },
      players: [],
    }));
  }, [gameState.currentRound.tokens]);

  // ✅ END ROUND (MUST BE AFTER startNextRound)
  const endRound = useCallback(() => {
    const { currentRound, players, config } = gameState;
    const [tokenA, tokenB] = currentRound.tokens;
    
    const updatedPlayers = players.map(player => {
      const profit = player.bets.reduce((total, bet) => {
        const priceChange = bet.tokenSide === 'A'
          ? (Math.random() - 0.4) * 0.1
          : (Math.random() - 0.4) * 0.1;
        return total + (bet.amount * priceChange);
      }, 0);
      
      return {
        ...player,
        profit,
        profitPercentage: player.totalBet > 0 ? (profit / player.totalBet) * 100 : 0,
      };
    });

    const winners = calculateRewards(updatedPlayers, currentRound.prizePool, config.rewardDistribution);
    
    const finalPlayers = updatedPlayers.map(player => {
      const winner = winners.find(w => w.wallet === player.wallet);
      return {
        ...player,
        rank: winner?.rank,
        reward: winner?.reward,
      };
    });

    const result: RoundResult = {
      roundId: currentRound.id,
      winner: currentRound.tokenAPool > currentRound.tokenBPool ? 'A' : 'B',
      winnerToken: currentRound.tokenAPool > currentRound.tokenBPool ? tokenA : tokenB,
      winners,
      totalPlayers: finalPlayers.length,
      totalPrizePool: currentRound.prizePool,
    };

    setGameState(prev => ({
      ...prev,
      currentRound: {
        ...prev.currentRound,
        status: 'results',
      },
      players: finalPlayers,
      roundHistory: [result, ...prev.roundHistory].slice(0, 10),
    }));

    setTimeout(() => {
      startNextRound();
    }, config.resultDuration * 1000);
  }, [gameState, config.resultDuration, startNextRound]);

  // ✅ SIMULATE AUTO TRADING
  const simulateAutoTrading = useCallback(() => {
    setGameState(prev => {
      const { currentRound, players } = prev;
      if (currentRound.status !== 'live') return prev;

      const updatedPlayers = players.map(player => {
        const fluctuation = (Math.random() - 0.4) * 0.02;
        const newProfit = player.profit + (player.totalBet * fluctuation);
        
        return {
          ...player,
          profit: newProfit,
          profitPercentage: player.totalBet > 0 ? (newProfit / player.totalBet) * 100 : 0,
        };
      });

      return { ...prev, players: updatedPlayers };
    });
  }, []);

  // ✅ ADD FAKE PLAYERS - FIXED: Pass 3 arguments
  const addFakePlayers = useCallback(() => {
    if (!config.fakePlayersEnabled) return;
    
    const fakePlayers = generateFakePlayers(
      config.fakePlayerCount,
      gameState.currentRound.tokens[0].symbol,
      gameState.currentRound.tokens[1].symbol
    );
    
    let tokenAPool = 0;
    let tokenBPool = 0;
    
    fakePlayers.forEach(player => {
      player.bets.forEach(bet => {
        const actualBet = calculateActualBet(bet.amount, config.platformFeePercent);
        if (bet.tokenSide === 'A') tokenAPool += actualBet;
        else tokenBPool += actualBet;
      });
    });

    setGameState(prev => ({
      ...prev,
      players: [...prev.players, ...fakePlayers],
      currentRound: {
        ...prev.currentRound,
        tokenAPool: prev.currentRound.tokenAPool + tokenAPool,
        tokenBPool: prev.currentRound.tokenBPool + tokenBPool,
        prizePool: prev.currentRound.prizePool + tokenAPool + tokenBPool,
      },
    }));
  }, [config, gameState.currentRound.tokens]);

  // ✅ PLACE BET
  const placeBet = useCallback((amount: number, tokenSide: 'A' | 'B'): boolean => {
    if (!userWallet) return false;
    if (amount < config.minBet) return false;
    
    const currentPlayer = gameState.players.find(p => p.wallet === userWallet);
    const currentTotalBet = currentPlayer?.totalBet || 0;
    
    if (currentTotalBet + amount > config.maxBetPerPlayer) return false;
    
    const actualBet = calculateActualBet(amount, config.platformFeePercent);
    const newBet: PlayerBet = {
      id: Math.random().toString(36).substring(2, 15),
      wallet: userWallet,
      amount: actualBet,
      tokenSide,
      timestamp: Date.now(),
    };

    setGameState(prev => {
      const existingPlayer = prev.players.find(p => p.wallet === userWallet);
      const newPlayers = existingPlayer
        ? prev.players.map(p => p.wallet === userWallet 
            ? { 
                ...p, 
                bets: [...p.bets, newBet],
                totalBet: p.totalBet + actualBet,
              }
            : p
          )
        : [{
            wallet: userWallet,
            bets: [newBet],
            totalBet: actualBet,
            profit: 0,
            profitPercentage: 0,
          }, ...prev.players];

      const newPool = tokenSide === 'A' 
        ? { tokenAPool: prev.currentRound.tokenAPool + actualBet, tokenBPool: prev.currentRound.tokenBPool }
        : { tokenAPool: prev.currentRound.tokenAPool, tokenBPool: prev.currentRound.tokenBPool + actualBet };

      return {
        ...prev,
        players: newPlayers,
        currentRound: {
          ...prev.currentRound,
          ...newPool,
          prizePool: prev.currentRound.prizePool + actualBet,
        },
      };
    });

    return true;
  }, [userWallet, config, gameState.players, gameState.currentRound]);

  // ✅ GET PLAYER BETS
  const getPlayerBets = useCallback((wallet: string): PlayerBet[] => {
    const player = gameState.players.find(p => p.wallet === wallet);
    return player?.bets || [];
  }, [gameState.players]);

  // ✅ GET LEADERBOARD
  const getLeaderboard = useCallback((): Player[] => {
    return sortPlayersByProfit(gameState.players);
  }, [gameState.players]);

  // ✅ GET ROUND PROGRESS
  const getCurrentRoundProgress = useCallback((): number => {
    const { currentRound } = gameState;
    if (!currentRound.startTime || !currentRound.endTime) return 0;
    
    const elapsed = currentTime - currentRound.startTime;
    const total = currentRound.endTime - currentRound.startTime;
    return Math.min(100, Math.max(0, (elapsed / total) * 100));
  }, [gameState.currentRound, currentTime]);

  const currentPlayer = userWallet 
    ? gameState.players.find(p => p.wallet === userWallet) || null
    : null;

  const timeLeft = calculateTimeLeft(gameState.currentRound.endTime, currentTime);

  const value: GameContextType = {
    gameState,
    currentTime,
    timeLeft,
    currentPlayer,
    placeBet,
    getPlayerBets,
    getLeaderboard,
    getCurrentRoundProgress,
  };

  return (
    <GameContext.Provider value={value}>
      {children}
    </GameContext.Provider>
  );
}

export const useGame = (): GameContextType => {
  const context = useContext(GameContext);
  if (!context) {
    throw new Error('useGame must be used within GameProvider');
  }
  return context;
};

export default GameContext;
